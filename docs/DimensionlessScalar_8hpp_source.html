41 /// \brief Abstract base class that represents any dimensionless scalar physical quantity. Such a
42 /// physical quantity is composed only of a value where the value is a scalar number. Such a
43 /// physical quantity has no unit of measure and no dimension set.
44 template <typename NumericType = double>
45 class DimensionlessScalar {
46  static_assert(std::is_floating_point<NumericType>::value,
47  "The NumericType template parameter of a physical quantity must be a numeric "
48  "floating-point type: float, double, or long double.");
49 
50 public:
51  /// \brief Physical dimension set of this physical quantity. Since this physical quantity is
52  /// dimensionless, its physical dimension set is simply the null set.
53  static constexpr PhQ::Dimensions Dimensions() {
54  return PhQ::Dimensionless;
55  }
56 
57  /// \brief Value of this physical quantity.

59  return value;
60  }

63  inline constexpr NumericType& MutableValue() noexcept {
64  return value;
65  }

67  /// \brief Sets the value of this physical quantity to the given value.
68  inline constexpr void SetValue(const NumericType value) noexcept {
69  this->value = value;
70  }

72  /// \brief Prints this physical quantity as a string.
73  [[nodiscard]] std::string Print() const {
74  return PhQ::Print(value);
75  }

107  /// \brief Copy constructor. Constructs a dimensionless scalar physical quantity by copying

111  : value(static_cast<NumericType>(other.Value())) {}
112 
113  /// \brief Move constructor. Constructs a dimensionless scalar physical quantity by moving another

116 
117  /// \brief Copy assignment operator. Assigns this dimensionless scalar physical quantity by
118  /// copying another one.
119  constexpr DimensionlessScalar<NumericType>& operator=(
120  const DimensionlessScalar<NumericType>& other) = default;
121 
122  /// \brief Copy assignment operator. Assigns this dimensionless scalar physical quantity by
123  /// copying another one.
124  template <typename OtherNumericType>
125  constexpr DimensionlessScalar<NumericType>& operator=(
126  const DimensionlessScalar<OtherNumericType>& other) {
127  value = static_cast<NumericType>(other.Value());

136  /// \brief Value of this physical quantity.
