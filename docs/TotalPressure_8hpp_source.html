46 /// \brief Total pressure, which is the sum of static pressure and dynamic pressure; see
47 /// PhQ::StaticPressure and PhQ::DynamicPressure. For total kinematic pressure, see
48 /// PhQ::TotalKinematicPressure.
49 template <typename NumericType = double>
50 class TotalPressure : public DimensionalScalar<Unit::Pressure, NumericType> {
51 public:
52  /// \brief Default constructor. Constructs a total pressure with an uninitialized value.
53  TotalPressure() = default;
54 
55  /// \brief Constructor. Constructs a total pressure with a given value expressed in a given
56  /// pressure unit.
57  TotalPressure(const NumericType value, const Unit::Pressure unit)
58  : DimensionalScalar<Unit::Pressure, NumericType>(value, unit) {}
59 
60  /// \brief Constructor. Constructs a total pressure from a given static pressure and dynamic
61  /// pressure using the definition of total pressure.
62  constexpr TotalPressure(const StaticPressure<NumericType>& static_pressure,
63  const DynamicPressure<NumericType>& dynamic_pressure)
64  : TotalPressure<NumericType>(static_pressure.Value() + dynamic_pressure.Value()) {}
65 
66  /// \brief Constructor. Constructs a total pressure from a given mass density and total kinematic

69  const TotalKinematicPressure<NumericType>& total_kinematic_pressure);
