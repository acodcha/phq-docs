53 /// \brief Dynamic viscosity, also known as molecular dynamic viscosity. Dynamic viscosity is the
54 /// relationship between the viscous stress of a material and its corresponding strain rate. Not to
55 /// be confused with kinematic viscosity, which is dynamic viscosity divided by mass density; see
56 /// PhQ::KinematicViscosity and PhQ::MassDensity. Also not to be confused with bulk dynamic
57 /// viscosity; see PhQ::BulkDynamicViscosity.
58 template <typename Number = double>
59 class DynamicViscosity : public DimensionalScalar<Unit::DynamicViscosity, Number> {
60 public:
61  /// \brief Default constructor. Constructs a dynamic viscosity with an uninitialized value.
62  DynamicViscosity() = default;
63 
64  /// \brief Constructor. Constructs a dynamic viscosity with a given value expressed in a given
65  /// dynamic viscosity unit.
66  DynamicViscosity(const Number value, const Unit::DynamicViscosity unit)
67  : DimensionalScalar<Unit::DynamicViscosity, Number>(value, unit) {}
68 
69  /// \brief Constructor. Constructs a dynamic viscosity from a given mass density and kinematic
70  /// viscosity using the definition of kinematic viscosity.
71  constexpr DynamicViscosity(const MassDensity<Number>& mass_density,
72  const KinematicViscosity<Number>& kinematic_viscosity)
73  : DynamicViscosity<Number>(mass_density.Value() * kinematic_viscosity.Value()) {}
74 
75  /// \brief Constructor. Constructs a dynamic viscosity from a given mass density, speed, length,

77  constexpr DynamicViscosity(

79  const Length<Number>& length, const ReynoldsNumber<Number>& reynolds_number);
