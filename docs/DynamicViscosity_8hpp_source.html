53 /// \brief Dynamic viscosity, also known as molecular dynamic viscosity. Dynamic viscosity is the
54 /// relationship between the viscous stress of a material and its corresponding strain rate. Not to
55 /// be confused with kinematic viscosity, which is dynamic viscosity divided by mass density.
56 template <typename Number = double>
57 class DynamicViscosity : public DimensionalScalar<Unit::DynamicViscosity, Number> {
58 public:
59  /// \brief Default constructor. Constructs a dynamic viscosity with an uninitialized value.
60  DynamicViscosity() = default;
61 
62  /// \brief Constructor. Constructs a dynamic viscosity with a given value expressed in a given
63  /// dynamic viscosity unit.
64  DynamicViscosity(const Number value, const Unit::DynamicViscosity unit)
65  : DimensionalScalar<Unit::DynamicViscosity, Number>(value, unit) {}
66 
67  /// \brief Constructor. Constructs a dynamic viscosity from a given mass density and kinematic
68  /// viscosity using the definition of kinematic viscosity.
69  constexpr DynamicViscosity(const MassDensity<Number>& mass_density,
70  const KinematicViscosity<Number>& kinematic_viscosity)
71  : DynamicViscosity<Number>(mass_density.Value() * kinematic_viscosity.Value()) {}
72

77  const Length<Number>& length, const ReynoldsNumber<Number>& reynolds_number);
