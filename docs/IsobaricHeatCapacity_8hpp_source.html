41 /// \brief Isobaric heat capacity, also known as heat capacity at constant pressure. For the
42 /// mass-specific isobaric heat capacity, see PhQ::SpecificIsobaricHeatCapacity.
43 template <typename Number = double>
44 class IsobaricHeatCapacity : public DimensionalScalar<Unit::HeatCapacity, Number> {
45 public:
46  /// \brief Default constructor. Constructs an isobaric heat capacity with an uninitialized value.
47  IsobaricHeatCapacity() = default;

49  /// \brief Constructor. Constructs an isobaric heat capacity with a given value expressed in a
50  /// given heat capacity unit.
51  IsobaricHeatCapacity(const Number value, const Unit::HeatCapacity unit)
52  : DimensionalScalar<Unit::HeatCapacity, Number>(value, unit) {}

54  /// \brief Constructor. Constructs an isobaric heat capacity from a given gas constant and

62  const GasConstant<Number>& gas_constant);

64  /// \brief Constructor. Constructs an isobaric heat capacity from a given isochoric heat capacity

66  constexpr IsobaricHeatCapacity(const HeatCapacityRatio<Number>& heat_capacity_ratio,
67  const IsochoricHeatCapacity<Number>& isochoric_heat_capacity)
