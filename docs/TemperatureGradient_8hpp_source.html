45 /// \brief Three-dimensional Euclidean temperature gradient vector. Contains three components in
46 /// Cartesian coordinates: x, y, and z. For a two-dimensional Euclidean temperature gradient vector
47 /// in the XY plane, see PhQ::PlanarTemperatureGradient. For scalar temperature gradient components
48 /// or for the magnitude of a temperature gradient vector, see PhQ::ScalarTemperatureGradient.
49 template <typename NumericType = double>
50 class TemperatureGradient : public DimensionalVector<Unit::TemperatureGradient, NumericType> {

59  : DimensionalVector<Unit::TemperatureGradient, NumericType>(value, unit) {}
