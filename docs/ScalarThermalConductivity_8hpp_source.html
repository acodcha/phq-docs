59 /// \brief Scalar component or resultant of a three-dimensional Euclidean thermal conductivity
60 /// symmetric dyadic tensor. In general, thermal conductivity is a tensor; however, in isotropic
61 /// materials, thermal conductivity simplifies to a scalar. For the related tensor, see
62 /// PhQ::ThermalConductivity.
63 template <typename Number = double>
64 class ScalarThermalConductivity : public DimensionalScalar<Unit::ThermalConductivity, Number> {
65 public:
66  /// \brief Default constructor. Constructs a scalar thermal conductivity with an uninitialized
67  /// value.
68  ScalarThermalConductivity() = default;

70  /// \brief Constructor. Constructs a scalar thermal conductivity with a given value expressed in a
71  /// given thermal conductivity unit.
72  ScalarThermalConductivity(const Number value, const Unit::ThermalConductivity unit)
73  : DimensionalScalar<Unit::ThermalConductivity, Number>(value, unit) {}
