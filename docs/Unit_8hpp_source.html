72 /// \brief Abstract class for converting a sequence of values expressed in a unit of measure to or
73 /// from the standard unit of measure of that type. Internal implementation detail not intended to
74 /// be used outside of the PhQ::ConvertInPlace, PhQ::Convert, and PhQ::ConvertStatically functions.
75 template <typename Unit, Unit UnitValue>
76 class Conversions {
77 public:
78  /// \brief Converts a sequence of values expressed in the standard unit of measure of a given unit
79  /// type to any given unit of measure of that type. Internal implementation detail not intended to
80  /// be used outside of the PhQ::ConvertInPlace, PhQ::Convert, and PhQ::ConvertStatically
81  /// functions.
82  template <typename NumericType>
83  static inline constexpr void FromStandard(NumericType* values, const std::size_t size) noexcept {
84  static_assert(std::is_floating_point<NumericType>::value,
85  "The NumericType template parameter of PhQ::Conversions::FromStandard must be a "
86  "numeric floating-point type: float, double, or long double.");
87  const NumericType* const end{values + size};
88  for (; values < end; ++values) {
89  Conversion<Unit, UnitValue>::FromStandard(*values);
90  }
91  }

93  /// \brief Converts a sequence of values expressed in any given unit of measure of a given unit
94  /// type to the standard unit of measure of that type. Internal implementation detail not intended

96  /// functions.
97  template <typename NumericType>
98  static inline constexpr void ToStandard(NumericType* values, const std::size_t size) noexcept {

127 /// \brief Converts a value expressed in a given unit of measure to a new unit of measure. The
128 /// conversion is performed in-place.
129 template <typename Unit, typename NumericType>
130 inline void ConvertInPlace(NumericType& value, const Unit original_unit, const Unit new_unit) {
131  static_assert(std::is_floating_point<NumericType>::value,
132  "The NumericType template parameter of PhQ::ConvertInPlace must be a numeric "
133  "floating-point type: float, double, or long double.");
134  if (original_unit != Standard<Unit>) {
135  Internal::MapOfConversionsToStandard<Unit, NumericType>.find(original_unit)->second(&value, 1);
136  }
137  if (new_unit != Standard<Unit>) {
138  Internal::MapOfConversionsFromStandard<Unit, NumericType>.find(new_unit)->second(&value, 1);
139  }
140 }

274 /// \brief Converts a value expressed in a given unit of measure to a new unit of measure. Returns
275 /// the converted value. The original value remains unchanged. This function can be evaluated at
276 /// compile time.
277 template <typename Unit, Unit OriginalUnit, Unit NewUnit, typename NumericType>
278 inline constexpr NumericType ConvertStatically(const NumericType value) {
279  static_assert(std::is_floating_point<NumericType>::value,
280  "The NumericType template parameter of PhQ::ConvertStatically must be a numeric "
281  "floating-point type: float, double, or long double.");
282  NumericType result{value};
283  Internal::Conversion<Unit, OriginalUnit>::ToStandard(result);
284  Internal::Conversion<Unit, NewUnit>::FromStandard(result);
285  return result;
