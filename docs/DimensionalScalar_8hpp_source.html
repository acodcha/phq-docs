42 /// \brief Abstract base class that represents any dimensional scalar physical quantity. Such a
43 /// physical quantity is composed of a value and a unit of measure where the value is a scalar
44 /// number.
45 template <typename UnitType, typename NumericType = double>
46 class DimensionalScalar {
47  static_assert(std::is_floating_point<NumericType>::value,
48  "The NumericType template parameter of a physical quantity must be a numeric "
49  "floating-point type: float, double, or long double.");
50 
51 public:
52  /// \brief Physical dimension set of this physical quantity.
53  static constexpr const PhQ::Dimensions& Dimensions() {
54  return PhQ::RelatedDimensions<UnitType>;
55  }
56 
57  /// \brief Standard unit of measure for this physical quantity. This physical quantity's value is
58  /// stored internally in this unit of measure.
59  static constexpr UnitType Unit() {
60  return PhQ::Standard<UnitType>;
61  }
62 
63  /// \brief Value of this physical quantity expressed in its standard unit of measure.
64  [[nodiscard]] constexpr NumericType Value() const noexcept {
65  return value;
66  }
67 
68  /// \brief Value of this physical quantity expressed in a given unit of measure.
69  [[nodiscard]] NumericType Value(const UnitType unit) const {
70  return PhQ::Convert(value, PhQ::Standard<UnitType>, unit);
71  }
72 
73  /// \brief Value of this physical quantity expressed in a given unit of measure. This method can
74  /// be evaluated statically at compile-time.
75  template <UnitType NewUnit>
76  [[nodiscard]] constexpr NumericType StaticValue() const {
77  return PhQ::ConvertStatically<UnitType, PhQ::Standard<UnitType>, NewUnit>(value);
78  }
79 
80  /// \brief Returns the value of this physical quantity expressed in its standard unit of measure
81  /// as a mutable value.
82  constexpr NumericType& MutableValue() noexcept {
83  return value;
84  }
85 
86  /// \brief Sets the value of this physical quantity expressed in its standard unit of measure to
87  /// the given value.
88  constexpr void SetValue(const NumericType value) noexcept {
89  this->value = value;
90  }
91 
92  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
93  /// in its standard unit of measure.
94  [[nodiscard]] std::string Print() const {
95  return PhQ::Print(value).append(" ").append(PhQ::Abbreviation(PhQ::Standard<UnitType>));
96  }
97 
98  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
99  /// in the given unit of measure.
100  [[nodiscard]] std::string Print(const UnitType unit) const {
101  return PhQ::Print(Value(unit)).append(" ").append(PhQ::Abbreviation(unit));
102  }
103 
104  /// \brief Serializes this physical quantity as a JSON message. This physical quantity's value is

107  return std::string{"{\"value\":"}
108  .append(PhQ::Print(value))

164 protected:
165  /// \brief Default constructor. Constructs a dimensional scalar physical quantity with an
166  /// uninitialized value.
167  DimensionalScalar() = default;
168 
169  /// \brief Constructor. Constructs a dimensional scalar physical quantity with a given value
170  /// expressed in its standard unit of measure.
171  explicit constexpr DimensionalScalar(const NumericType value) : value(value) {}
172 
173  /// \brief Constructor. Constructs a dimensional scalar physical quantity with a given value
174  /// expressed in a given unit of measure.
175  DimensionalScalar(const NumericType value, const UnitType unit) : value(value) {
176  PhQ::ConvertInPlace(this->value, unit, PhQ::Standard<UnitType>);
177  }

187  /// one.
188  template <typename OtherNumericType>

190  : value(static_cast<NumericType>(other.Value())) {}

193  /// one.
194  constexpr DimensionalScalar(DimensionalScalar<UnitType, NumericType>&& other) noexcept = default;
195 
196  /// \brief Copy assignment operator. Assigns this dimensional scalar physical quantity by copying
197  /// another one.
198  constexpr DimensionalScalar<UnitType, NumericType>& operator=(
199  const DimensionalScalar<UnitType, NumericType>& other) = default;
200 
201  /// \brief Copy assignment operator. Assigns this dimensional scalar physical quantity by copying
202  /// another one.
203  template <typename OtherNumericType>
204  constexpr DimensionalScalar<UnitType, NumericType>& operator=(
205  const DimensionalScalar<UnitType, OtherNumericType>& other) {
206  value = static_cast<NumericType>(other.Value());
207  return *this;
208  }
209 
210  /// \brief Move assignment operator. Assigns this dimensional scalar physical quantity by moving
211  /// another one.
212  constexpr DimensionalScalar<UnitType, NumericType>& operator=(
213  DimensionalScalar<UnitType, NumericType>&& other) noexcept = default;
214 
215  /// \brief Value of this physical quantity expressed in its standard unit of measure.
216  NumericType value;
