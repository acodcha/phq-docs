42 /// \brief Abstract base class that represents any dimensional scalar physical quantity. Such a
43 /// physical quantity is composed of a value and a unit of measure where the value is a scalar
44 /// number.
45 template <typename UnitType, typename Number = double>
46 class DimensionalScalar {
47  static_assert(
48  std::is_floating_point<Number>::value,
49  "The Number template parameter of a physical quantity must be a floating-point number type.");
50 
51 public:
52  /// \brief Physical dimension set of this physical quantity.
53  static constexpr const PhQ::Dimensions& Dimensions() {
54  return PhQ::RelatedDimensions<UnitType>;
55  }
56 
57  /// \brief Standard unit of measure for this physical quantity. This physical quantity's value is
58  /// stored internally in this unit of measure.
59  static constexpr UnitType Unit() {
60  return PhQ::Standard<UnitType>;
61  }
62 
63  /// \brief Value of this physical quantity expressed in its standard unit of measure.
64  [[nodiscard]] constexpr Number Value() const noexcept {
65  return value;
66  }
67 
68  /// \brief Value of this physical quantity expressed in a given unit of measure.
69  [[nodiscard]] Number Value(const UnitType unit) const {
70  Number result{value};
71  PhQ::Convert(result, PhQ::Standard<UnitType>, unit);
72  return result;
73  }
74 
75  /// \brief Value of this physical quantity expressed in a given unit of measure. This method can
76  /// be evaluated statically at compile-time.
77  template <UnitType NewUnit>
78  [[nodiscard]] constexpr Number StaticValue() const {
79  return PhQ::StaticConvertCopy<UnitType, PhQ::Standard<UnitType>, NewUnit>(value);
80  }

82  /// \brief Returns the value of this physical quantity expressed in its standard unit of measure
