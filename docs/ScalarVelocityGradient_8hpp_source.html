42 /// \brief Scalar component or resultant of a three-dimensional Euclidean velocity gradient dyadic
43 /// tensor. For the related tensor, see PhQ::VelocityGradient. Can also represent the time rate of
44 /// change of a scalar displacement gradient; see PhQ::ScalarDisplacementGradient, PhQ::Time, and
45 /// PhQ::Frequency.
46 template <typename Number = double>
47 class ScalarVelocityGradient : public DimensionalScalar<Unit::Frequency, Number> {
48 public:

53  /// given frequency unit.

57  /// \brief Constructor. Constructs a scalar velocity gradient from a given scalar displacement
