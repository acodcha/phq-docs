69 /// \brief Static pressure. Pressure of a fluid at rest. Not to be confused with dynamic pressure,
70 /// which is the additional pressure resulting from the kinetic energy of a flowing fluid, or total
71 /// pressure, which is the sum of static pressure and dynamic pressure. Can represent either an
72 /// absolute static pressure or a static pressure difference relative to another static pressure.
73 template <typename Number = double>
74 class StaticPressure : public DimensionalScalar<Unit::Pressure, Number> {
75 public:
76  /// \brief Default constructor. Constructs a static pressure with an uninitialized value.
77  StaticPressure() = default;
78

80  /// pressure unit.

82  : DimensionalScalar<Unit::Pressure, Number>(value, unit) {}
