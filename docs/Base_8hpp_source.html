102 /// \brief Attempts to parse some given text into an enumeration. Returns the enumeration if one is
103 /// found, or std::nullopt otherwise. For example, PhQ::Parse<PhQ::Unit::Time>("hr") returns
104 /// PhQ::Unit::Time::Hour.
105 template <typename Enumeration>
106 std::optional<Enumeration> Parse(const std::string_view spelling) {
107  const typename std::unordered_map<std::string_view, Enumeration>::const_iterator found{
108  Internal::Spellings<Enumeration>.find(spelling)};
109  if (found != Internal::Spellings<Enumeration>.cend()) {
110  return found->second;
111  }
112  return std::nullopt;
113 }

167 /// \brief Prints a given floating-point number as a string. Prints enough digits to represent the
168 /// number exactly. The printed number of digits depends on the type of the floating-point number.
169 template <typename Number>
170 [[nodiscard]] inline std::string Print(const Number value) {
171  static_assert(
172  std::is_floating_point<Number>::value,
173  "The Number template parameter of PhQ::Print<Number> must be a floating-point number type.");
174  const Number absolute{std::abs(value)};
175  std::ostringstream stream;
176  if (absolute < 1.0) {
177  // Interval: [0, 1[
178  if (absolute < 0.001) {
179  // Interval: [0, 0.001[
180  if (absolute == 0.0) {
181  // Interval: [0, 0]
182  stream << 0;
183  } else {
184  // Interval: ]0, 0.001[
185  stream << std::scientific << std::setprecision(std::numeric_limits<Number>::max_digits10)
186  << value;
187  }
188  } else {
189  // Interval: [0.001, 1[
190  if (absolute < 0.1) {
191  // Interval: [0.001, 0.1[
192  if (absolute < 0.01) {
193  // Interval: [0.001, 0.01[
194  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 + 3)
195  << value;
196  } else {
197  // Interval: [0.01, 0.1[
198  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 + 2)
199  << value;
200  }
201  } else {
202  // Interval: [0.1, 1[
203  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 + 1)
204  << value;
205  }
206  }
207  } else {
208  // Interval: [1, +inf[
209  if (absolute < 1000.0) {
210  // Interval: [1, 1000[
211  if (absolute < 10.0) {
212  // Interval: [1, 10[
213  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10)
214  << value;
215  } else {
216  // Interval: [10, 1000[
217  if (absolute < 100.0) {
218  // Interval: [10, 100[
219  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 - 1)
220  << value;
221  } else {
222  // Interval: [100, 1000[
223  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 - 2)
224  << value;
225  }
226  }
227  } else {
228  // Interval: [1000, +inf[
229  if (absolute < 10000.0) {
230  // Interval: [1000, 10000[
231  stream << std::fixed << std::setprecision(std::numeric_limits<Number>::max_digits10 - 3)
232  << value;
233  } else {
234  // Interval: [10000, +inf[
235  stream << std::scientific << std::setprecision(std::numeric_limits<Number>::max_digits10)
236  << value;
237  }
238  }
239  }
240  return stream.str();
241 }
