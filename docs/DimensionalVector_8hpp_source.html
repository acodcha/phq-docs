42 /// \brief Abstract base class that represents any dimensional vector physical quantity. Such a
43 /// physical quantity is composed of a value and a unit of measure where the value is a
44 /// three-dimensional vector.
45 template <typename UnitType, typename NumericType = double>
46 class DimensionalVector {
47 public:
48  /// \brief Physical dimension set of this physical quantity.
49  static constexpr const PhQ::Dimensions& Dimensions() {
50  return PhQ::RelatedDimensions<UnitType>;
51  }
52 
53  /// \brief Standard unit of measure for this physical quantity. This physical quantity's value is
54  /// stored internally in this unit of measure.
55  static constexpr UnitType Unit() {
56  return PhQ::Standard<UnitType>;
57  }
58 
59  /// \brief Value of this physical quantity expressed in its standard unit of measure.
60  [[nodiscard]] constexpr const PhQ::Vector<NumericType>& Value() const noexcept {
61  return value;
62  }
63 
64  /// \brief Value of this physical quantity expressed in a given unit of measure.
65  [[nodiscard]] PhQ::Vector<NumericType> Value(const UnitType unit) const {
66  return PhQ::Convert(value, PhQ::Standard<UnitType>, unit);
67  }
68 
69  /// \brief Value of this physical quantity expressed in a given unit of measure. This method can
70  /// be evaluated statically at compile-time.
71  template <UnitType NewUnit>
72  [[nodiscard]] constexpr PhQ::Vector<NumericType> StaticValue() const {
73  return PhQ::ConvertStatically<UnitType, PhQ::Standard<UnitType>, NewUnit>(value);
74  }
75 
76  /// \brief Returns the value of this physical quantity expressed in its standard unit of measure
77  /// as a mutable value.
78  constexpr PhQ::Vector<NumericType>& MutableValue() noexcept {
79  return value;
80  }
81 
82  /// \brief Sets the value of this physical quantity expressed in its standard unit of measure to
83  /// the given value.
84  constexpr void SetValue(const PhQ::Vector<NumericType>& value) noexcept {
85  this->value = value;
86  }
87 
88  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
89  /// in its standard unit of measure.
90  [[nodiscard]] std::string Print() const {
91  return value.Print().append(" ").append(PhQ::Abbreviation(PhQ::Standard<UnitType>));
92  }
93 
94  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
95  /// in the given unit of measure.
96  [[nodiscard]] std::string Print(const UnitType unit) const {
97  return Value(unit).Print().append(" ").append(PhQ::Abbreviation(unit));
98  }
99 
100  /// \brief Serializes this physical quantity as a JSON message. This physical quantity's value is
101  /// expressed in its standard unit of measure.
102  [[nodiscard]] std::string JSON() const {
103  return std::string{"{\"value\":"}
104  .append(value.JSON())
105  .append(R"(,"unit":")")
106  .append(PhQ::Abbreviation(PhQ::Standard<UnitType>))
107  .append("\"}");
108  }
109 
110  /// \brief Serializes this physical quantity as a JSON message. This physical quantity's value is
111  /// expressed in the given unit of measure.
112  [[nodiscard]] std::string JSON(const UnitType unit) const {
113  return std::string{"{\"value\":"}
114  .append(Value(unit).JSON())
115  .append(R"(,"unit":")")
116  .append(PhQ::Abbreviation(unit))
117  .append("\"}");
118  }
119 
120  /// \brief Serializes this physical quantity as an XML message. This physical quantity's value is

124  .append(value.XML())

150  /// \brief Serializes this physical quantity as a YAML message. This physical quantity's value is

160 protected:
161  /// \brief Default constructor. Constructs a dimensional vector physical quantity with an
162  /// uninitialized value.
163  DimensionalVector() = default;
164 
165  /// \brief Constructor. Constructs a dimensional vector physical quantity with a given value
166  /// expressed in its standard unit of measure.
167  explicit constexpr DimensionalVector(const PhQ::Vector<NumericType>& value) : value(value) {}
168 
169  /// \brief Constructor. Constructs a dimensional vector physical quantity with a given value
170  /// expressed in a given unit of measure.
171  DimensionalVector(const PhQ::Vector<NumericType>& value, const UnitType unit) : value(value) {
172  PhQ::ConvertInPlace(this->value, unit, PhQ::Standard<UnitType>);
173  }
174 
175  /// \brief Destructor. Destroys this dimensional vector physical quantity.
176  ~DimensionalVector() noexcept = default;
177 
178  /// \brief Copy constructor. Constructs a dimensional vector physical quantity by copying another
179  /// one.
180  constexpr DimensionalVector(const DimensionalVector<UnitType, NumericType>& other) = default;
181 
182  /// \brief Copy constructor. Constructs a dimensional vector physical quantity by copying another
183  /// one.
184  template <typename OtherNumericType>
185  explicit constexpr DimensionalVector(const DimensionalVector<UnitType, OtherNumericType>& other)
186  : value(static_cast<PhQ::Vector<NumericType>>(other.Value())) {}
187 
188  /// \brief Move constructor. Constructs a dimensional vector physical quantity by moving another
189  /// one.
190  constexpr DimensionalVector(DimensionalVector<UnitType, NumericType>&& other) noexcept = default;
191 
192  /// \brief Copy assignment operator. Assigns this dimensional vector physical quantity by copying
193  /// another one.
194  constexpr DimensionalVector<UnitType, NumericType>& operator=(
195  const DimensionalVector<UnitType, NumericType>& other) = default;
196 
197  /// \brief Copy assignment operator. Assigns this dimensional vector physical quantity by copying
198  /// another one.
199  template <typename OtherNumericType>
200  constexpr DimensionalVector<UnitType, NumericType>& operator=(
201  const DimensionalVector<UnitType, OtherNumericType>& other) {
202  value = static_cast<PhQ::Vector<NumericType>>(other.Value());
203  return *this;
204  }
205 
206  /// \brief Move assignment operator. Assigns this dimensional vector physical quantity by moving
207  /// another one.
208  constexpr DimensionalVector<UnitType, NumericType>& operator=(
209  DimensionalVector<UnitType, NumericType>&& other) noexcept = default;
210 
211  /// \brief Value of this physical quantity expressed in its standard unit of measure.
212  PhQ::Vector<NumericType> value;
213 };
