46 /// \brief Three-dimensional Euclidean displacement gradient dyadic tensor. Gradient of the
47 /// displacement vector. May be symmetric or asymmetric. Contains nine components in Cartesian
48 /// coordinates: xx, xy, xz, yx, yy, yz, zx, zy, and zz. For the scalar components or resultants of
49 /// a displacement gradient tensor, see PhQ::ScalarDisplacementGradient. The time rate of change of
50 /// a displacement gradient is a velocity gradient; see PhQ::VelocityGradient, PhQ::Time, and
51 /// PhQ::Frequency.
52 template <typename Number = double>
53 class DisplacementGradient : public DimensionlessDyad<Number> {
54 public:
55  /// \brief Default constructor. Constructs a displacement gradient tensor with an uninitialized
56  /// value.
57  DisplacementGradient() = default;
58 
59  /// \brief Constructor. Constructs a displacement gradient tensor whose value has the given xx,
60  /// xy, xz, yx, yy, yz, zx, zy, and zz Cartesian components.
61  constexpr DisplacementGradient(
62  const Number xx, const Number xy, const Number xz, const Number yx, const Number yy,
63  const Number yz, const Number zx, const Number zy, const Number zz)
64  : DimensionlessDyad<Number>(xx, xy, xz, yx, yy, yz, zx, zy, zz) {}

66  /// \brief Constructor. Constructs a displacement gradient tensor from a given array representing
67  /// its value's xx, xy, xz, yx, yy, yz, zx, zy, and zz Cartesian components.
68  explicit constexpr DisplacementGradient(const std::array<Number, 9>& xx_xy_xz_yx_yy_yz_zx_zy_zz)
69  : DimensionlessDyad<Number>(xx_xy_xz_yx_yy_yz_zx_zy_zz) {}
