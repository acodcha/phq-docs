222  /// \brief Sets the value of this direction by normalizing the given x, y, and z Cartesian
223  /// components to a unit vector. If x = 0, y = 0, and z = 0, sets the direction to the zero
224  /// vector.
225  constexpr void Set(const NumericType x, const NumericType y, const NumericType z) {
226  const NumericType magnitude_squared{x * x + y * y + z * z};
227  if (magnitude_squared > static_cast<NumericType>(0)) {
228  const NumericType magnitude{std::sqrt(magnitude_squared)};
229  this->value = Vector{x / magnitude, y / magnitude, z / magnitude};
230  } else {
231  this->value = Vector<>::Zero();
232  }
233  }
234 
235  /// \brief Sets the value of this direction by normalizing the given x, y, and z Cartesian
236  /// components to a unit vector. If x = 0, y = 0, and z = 0, sets the direction to the zero
237  /// vector.
238  constexpr void Set(const std::array<NumericType, 3>& x_y_z) {
239  const NumericType magnitude_squared{
240  x_y_z[0] * x_y_z[0] + x_y_z[1] * x_y_z[1] + x_y_z[2] * x_y_z[2]};
241  if (magnitude_squared > static_cast<NumericType>(0)) {
242  const NumericType magnitude{std::sqrt(magnitude_squared)};
243  this->value = Vector{x_y_z[0] / magnitude, x_y_z[1] / magnitude, x_y_z[2] / magnitude};
244  } else {
245  this->value = Vector<>::Zero();
246  }
247  }

252  Set(value.x_y_z());
