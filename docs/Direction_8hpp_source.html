113 /// \brief Three-dimensional Euclidean direction vector. Contains three components in Cartesian

221  /// \brief Sets the value of this direction by normalizing the given x, y, and z Cartesian
222  /// components to a unit vector. If x = 0, y = 0, and z = 0, sets the direction to the zero
223  /// vector.
224  constexpr void Set(const Number x, const Number y, const Number z) {
225  const Number magnitude_squared{x * x + y * y + z * z};
226  if (magnitude_squared > static_cast<Number>(0)) {
227  const Number magnitude{std::sqrt(magnitude_squared)};
228  this->value = Vector{x / magnitude, y / magnitude, z / magnitude};
229  } else {
230  this->value = Vector<>::Zero();
231  }
232  }
233 
234  /// \brief Sets the value of this direction by normalizing the given x, y, and z Cartesian
235  /// components to a unit vector. If x = 0, y = 0, and z = 0, sets the direction to the zero
236  /// vector.
237  constexpr void Set(const std::array<Number, 3>& x_y_z) {
238  const Number magnitude_squared{x_y_z[0] * x_y_z[0] + x_y_z[1] * x_y_z[1] + x_y_z[2] * x_y_z[2]};
239  if (magnitude_squared > static_cast<Number>(0)) {
240  const Number magnitude{std::sqrt(magnitude_squared)};
241  this->value = Vector{x_y_z[0] / magnitude, x_y_z[1] / magnitude, x_y_z[2] / magnitude};
242  } else {
243  this->value = Vector<>::Zero();
244  }
245  }

249  constexpr void Set(const Vector<Number>& value) {

251  }
