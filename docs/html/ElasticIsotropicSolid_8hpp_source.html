379  /// \brief Returns the stress resulting from a given strain.
380  [[nodiscard]] inline PhQ::Stress<float> Stress(const PhQ::Strain<float>& strain) const override {
381  // stress = a * strain + b * trace(strain) * identity_matrix
382  // a = 2 * shear_modulus
383  // b = lame_first_modulus
384  const float temporary{static_cast<float>(lame_first_modulus.Value())
385  * static_cast<float>(strain.Value().Trace())};
386  return {
387  static_cast<float>(2) * static_cast<float>(shear_modulus.Value())
388  * static_cast<SymmetricDyad<float>>(strain.Value())
389  + SymmetricDyad<float>{temporary, static_cast<float>(0), static_cast<float>(0),
390  temporary, static_cast<float>(0), temporary},
391  Standard<Unit::Pressure>
392  };
393  }

454  /// \brief Returns the strain resulting from a given stress.
455  [[nodiscard]] inline PhQ::Strain<float> Strain(const PhQ::Stress<float>& stress) const override {
456  // strain = a * stress + b * trace(stress) * identity_matrix
457  // a = 1 / (2 * shear_modulus)
458  // b = -1 * lame_first_modulus / (2 * shear_modulus * (2 * shear_modulus + 3
459  // * lame_first_modulus))
460  const float a{static_cast<float>(1)
461  / (static_cast<float>(2) * static_cast<float>(shear_modulus.Value()))};
462  const float b{-static_cast<float>(lame_first_modulus.Value())
463  / (static_cast<float>(2) * static_cast<float>(shear_modulus.Value())
464  * (static_cast<float>(2) * static_cast<float>(shear_modulus.Value())
465  + static_cast<float>(3) * static_cast<float>(lame_first_modulus.Value())))};

478  // a = 1 / (2 * shear_modulus)

480  // * lame_first_modulus))

487  + static_cast<double>(3) * static_cast<double>(lame_first_modulus.Value())))};
