43 /// \brief Abstract base class that represents any dimensionless dyadic tensor physical quantity.
44 /// Such a physical quantity is composed only of a value where the value is a three-dimensional
45 /// dyadic tensor. The tensor may be non-symmetric. Such a physical quantity has no unit of measure
46 /// and no dimension set.
47 template <typename Number = double>
48 class DimensionlessDyad {
49  static_assert(
50  std::is_floating_point<Number>::value,
51  "The Number template parameter of a physical quantity must be a floating-point number type.");
52 
53 public:
54  /// \brief Physical dimension set of this physical quantity. Since this physical quantity is
55  /// dimensionless, its physical dimension set is simply the null set.
56  static constexpr PhQ::Dimensions Dimensions() {
57  return PhQ::Dimensionless;
58  }
