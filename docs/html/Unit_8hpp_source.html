75 class Conversions {
76 public:

79  /// be used outside of the PhQ::Convert, PhQ::ConvertCopy, and PhQ::StaticConvertCopy functions.
80  template <typename Number>

124 /// \brief Converts a value expressed in a given unit of measure to a new unit of measure. The
125 /// conversion is performed in-place.
126 template <typename Unit, typename Number>
127 inline void Convert(Number& value, const Unit original_unit, const Unit new_unit) {
128  static_assert(
129  std::is_floating_point<Number>::value,
130  "The Number template parameter of PhQ::Convert must be a floating-point number type.");
131  if (original_unit != Standard<Unit>) {
132  Internal::MapOfConversionsToStandard<Unit, Number>.find(original_unit)->second(&value, 1);
133  }
134  if (new_unit != Standard<Unit>) {
135  Internal::MapOfConversionsFromStandard<Unit, Number>.find(new_unit)->second(&value, 1);
136  }
137 }

252 template <typename Unit, Unit OriginalUnit, Unit NewUnit, typename Number>
253 inline constexpr Number StaticConvertCopy(const Number value) {
