218  /// \brief Sets the value of this planar direction by normalizing the given x and y Cartesian
219  /// components to a unit planar vector. If x = 0 and y = 0, sets the planar direction to the zero
220  /// planar vector.
221  constexpr void Set(const Number x, const Number y) {
222  const Number magnitude_squared{x * x + y * y};
223  if (magnitude_squared > static_cast<Number>(0)) {
224  const Number magnitude{std::sqrt(magnitude_squared)};
225  this->value = PlanarVector{x / magnitude, y / magnitude};
226  } else {
227  this->value = PlanarVector<>::Zero();
228  }
229  }

231  /// \brief Sets the value of this planar direction by normalizing the given x and y Cartesian
232  /// components to a unit planar vector. If x = 0 and y = 0, sets the planar direction to the zero
233  /// planar vector.
234  constexpr void Set(const std::array<Number, 2>& x_y) {
235  const Number magnitude_squared{x_y[0] * x_y[0] + x_y[1] * x_y[1]};
236  if (magnitude_squared > static_cast<Number>(0)) {
237  const Number magnitude{std::sqrt(magnitude_squared)};
238  this->value = PlanarVector{x_y[0] / magnitude, x_y[1] / magnitude};
239  } else {
240  this->value = PlanarVector<>::Zero();
241  }
242  }
