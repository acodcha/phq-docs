47 /// \brief Constitutive model for an incompressible Newtonian fluid. This is the simplest
48 /// constitutive model for a fluid. The viscous stress tensor at a point is a linear function of
49 /// only the local strain rate tensor at that point.
50 template <typename Number = double>
51 class ConstitutiveModel::IncompressibleNewtonianFluid : public ConstitutiveModel {
