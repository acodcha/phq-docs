42 /// \brief Abstract base class that represents any dimensional symmetric dyadic tensor physical
43 /// quantity. Such a physical quantity is composed of a value and a unit of measure where the value
44 /// is a three-dimensional symmetric dyadic tensor.
45 template <typename UnitType, typename NumericType = double>
46 class DimensionalSymmetricDyad {
47 public:
48  /// \brief Physical dimension set of this physical quantity.
49  static constexpr const PhQ::Dimensions& Dimensions() {
50  return PhQ::RelatedDimensions<UnitType>;
51  }
52 
53  /// \brief Standard unit of measure for this physical quantity. This physical quantity's value is
54  /// stored internally in this unit of measure.
55  static constexpr UnitType Unit() {
56  return PhQ::Standard<UnitType>;
57  }
58 
59  /// \brief Value of this physical quantity expressed in its standard unit of measure.
60  [[nodiscard]] constexpr const PhQ::SymmetricDyad<NumericType>& Value() const noexcept {
61  return value;
62  }
63 
64  /// \brief Value of this physical quantity expressed in a given unit of measure.
65  [[nodiscard]] PhQ::SymmetricDyad<NumericType> Value(const UnitType unit) const {
66  return PhQ::Convert(value, PhQ::Standard<UnitType>, unit);
67  }
68 
69  /// \brief Value of this physical quantity expressed in a given unit of measure. This method can
70  /// be evaluated statically at compile-time.
71  template <UnitType NewUnit>
72  [[nodiscard]] constexpr PhQ::SymmetricDyad<NumericType> StaticValue() const {
73  return PhQ::ConvertStatically<UnitType, PhQ::Standard<UnitType>, NewUnit>(value);
74  }
75 
76  /// \brief Returns the value of this physical quantity expressed in its standard unit of measure
77  /// as a mutable value.
78  constexpr PhQ::SymmetricDyad<NumericType>& MutableValue() noexcept {
79  return value;
80  }
81 
82  /// \brief Sets the value of this physical quantity expressed in its standard unit of measure to
83  /// the given value.
84  constexpr void SetValue(const PhQ::SymmetricDyad<NumericType>& value) noexcept {
85  this->value = value;
86  }

88  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
89  /// in its standard unit of measure.
90  [[nodiscard]] std::string Print() const {
91  return value.Print().append(" ").append(PhQ::Abbreviation(PhQ::Standard<UnitType>));
92  }
93 
94  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
95  /// in the given unit of measure.
96  [[nodiscard]] std::string Print(const UnitType unit) const {
97  return Value(unit).Print().append(" ").append(PhQ::Abbreviation(unit));
98  }

217  PhQ::SymmetricDyad<NumericType> value;
