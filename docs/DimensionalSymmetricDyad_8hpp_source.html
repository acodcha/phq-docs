42 /// \brief Abstract base class that represents any dimensional symmetric dyadic tensor physical
43 /// quantity. Such a physical quantity is composed of a value and a unit of measure where the value
44 /// is a three-dimensional symmetric dyadic tensor.
45 template <typename UnitType, typename Number = double>
46 class DimensionalSymmetricDyad {
47 public:
48  /// \brief Physical dimension set of this physical quantity.
49  static constexpr const PhQ::Dimensions& Dimensions() {
50  return PhQ::RelatedDimensions<UnitType>;
51  }
52 
53  /// \brief Standard unit of measure for this physical quantity. This physical quantity's value is
54  /// stored internally in this unit of measure.
55  static constexpr UnitType Unit() {
56  return PhQ::Standard<UnitType>;
57  }
58 
59  /// \brief Value of this physical quantity expressed in its standard unit of measure.
60  [[nodiscard]] constexpr const PhQ::SymmetricDyad<Number>& Value() const noexcept {
61  return value;
62  }
63 
64  /// \brief Value of this physical quantity expressed in a given unit of measure.
65  [[nodiscard]] PhQ::SymmetricDyad<Number> Value(const UnitType unit) const {
66  PhQ::SymmetricDyad<Number> result{value};
67  PhQ::Convert(result, PhQ::Standard<UnitType>, unit);
68  return result;
69  }
70 
71  /// \brief Value of this physical quantity expressed in a given unit of measure. This method can
72  /// be evaluated statically at compile-time.
73  template <UnitType NewUnit>
74  [[nodiscard]] constexpr PhQ::SymmetricDyad<Number> StaticValue() const {
75  return PhQ::StaticConvertCopy<UnitType, PhQ::Standard<UnitType>, NewUnit>(value);
76  }
77 
78  /// \brief Returns the value of this physical quantity expressed in its standard unit of measure
79  /// as a mutable value.
80  constexpr PhQ::SymmetricDyad<Number>& MutableValue() noexcept {
81  return value;
82  }
83 
84  /// \brief Sets the value of this physical quantity expressed in its standard unit of measure to
85  /// the given value.
86  constexpr void SetValue(const PhQ::SymmetricDyad<Number>& value) noexcept {
87  this->value = value;
88  }

90  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed
91  /// in its standard unit of measure.
92  [[nodiscard]] std::string Print() const {
93  return value.Print().append(" ").append(PhQ::Abbreviation(PhQ::Standard<UnitType>));

96  /// \brief Prints this physical quantity as a string. This physical quantity's value is expressed

219  PhQ::SymmetricDyad<Number> value;
