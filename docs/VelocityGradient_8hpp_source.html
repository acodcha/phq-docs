46 /// \brief Three-dimensional Euclidean velocity gradient dyadic tensor. Gradient of the velocity
47 /// vector. May be symmetric or asymmetric. Contains nine components in Cartesian coordinates: xx,
48 /// xy, xz, yx, yy, yz, zx, zy, and zz. For the scalar components or resultants of a velocity
49 /// gradient tensor, see PhQ::ScalarVelocityGradient. Can also represent the time rate of change of
50 /// a displacement gradient; see PhQ::DisplacementGradient, PhQ::Time, and PhQ::Frequency.
51 template <typename Number = double>
52 class VelocityGradient : public DimensionalDyad<Unit::Frequency, Number> {
53 public:

58  /// given frequency unit.
59  VelocityGradient(const Dyad<Number>& value, const Unit::Frequency& unit)
60  : DimensionalDyad<Unit::Frequency, Number>(value, unit) {}
