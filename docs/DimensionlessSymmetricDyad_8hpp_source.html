41 /// \brief Abstract base class that represents any dimensionless symmetric dyadic tensor physical
42 /// quantity. Such a physical quantity is composed only of a value where the value is a
43 /// three-dimensional symmetric dyadic tensor. Such a physical quantity has no unit of measure and
44 /// no dimension set.
45 template <typename NumericType = double>
46 class DimensionlessSymmetricDyad {
47  static_assert(std::is_floating_point<NumericType>::value,
48  "The NumericType template parameter of a physical quantity must be a numeric "
49  "floating-point type: float, double, or long double.");
50 
51 public:
52  /// \brief Physical dimension set of this physical quantity. Since this physical quantity is
53  /// dimensionless, its physical dimension set is simply the null set.
54  [[nodiscard]] static constexpr PhQ::Dimensions Dimensions() {
55  return PhQ::Dimensionless;
56  }
57

60  return value;
