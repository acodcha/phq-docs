43 /// \brief Abstract base class that represents any dimensionless symmetric dyadic tensor physical
44 /// quantity. Such a physical quantity is composed only of a value where the value is a
45 /// three-dimensional symmetric dyadic tensor. Such a physical quantity has no unit of measure and
46 /// no dimension set.
47 template <typename Number = double>
48 class DimensionlessSymmetricDyad {
49  static_assert(
50  std::is_floating_point<Number>::value,
51  "The Number template parameter of a physical quantity must be a floating-point number type.");
