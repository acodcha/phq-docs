39 /// \brief Bulk dynamic viscosity, also known as volume dynamic viscosity or dilatational dynamic
40 /// viscosity. Not to be confused with dynamic viscosity; see PhQ::DynamicViscosity.
41 template <typename Number = double>
42 class BulkDynamicViscosity : public DimensionalScalar<Unit::DynamicViscosity, Number> {
