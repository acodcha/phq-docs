57 /// \brief Scalar traction component or magnitude of a traction vector. Traction is similar to
58 /// pressure; however, traction can act in any direction, whereas pressure always acts compressively
59 /// perpendicular to a surface. For a three-dimensional Euclidean traction vector, see
60 /// PhQ::Traction. For a two-dimensional Euclidean traction vector in the XY plane, see
61 /// PhQ::PlanarTraction.
62 template <typename Number = double>
63 class ScalarTraction : public DimensionalScalar<Unit::Pressure, Number> {
64 public:
65  /// \brief Default constructor. Constructs a scalar traction with an uninitialized value.
66  ScalarTraction() = default;
67 
68  /// \brief Constructor. Constructs a scalar traction with a given value expressed in a given
69  /// pressure unit.
70  ScalarTraction(const Number value, const Unit::Pressure unit)
71  : DimensionalScalar<Unit::Pressure, Number>(value, unit) {}
