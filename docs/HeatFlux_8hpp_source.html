58  /// \brief Constructor. Constructs a heat flux vector from a given scalar heat flux magnitude and
59  /// direction.
60  constexpr HeatFlux(

64  /// \brief Constructor. Constructs a heat flux vector from a given scalar thermal conductivity and
65  /// temperature gradient vector using Fourier's law of heat conduction. Since heat flows opposite
66  /// the temperature gradient, the resulting heat flux direction is opposite the temperature
67  /// gradient direction.
68  constexpr HeatFlux(const ScalarThermalConductivity<Number>& scalar_thermal_conductivity,
69  const TemperatureGradient<Number>& temperature_gradient)
70  : HeatFlux<Number>(-scalar_thermal_conductivity.Value() * temperature_gradient.Value()) {}
71 
72  /// \brief Constructor. Constructs a heat flux vector from a given thermal conductivity tensor and
73  /// temperature gradient vector using Fourier's law of heat conduction. Since heat flows opposite
74  /// the temperature gradient, the resulting heat flux direction is opposite the temperature
75  /// gradient direction.
76  constexpr HeatFlux(const ThermalConductivity<Number>& thermal_conductivity,
77  const TemperatureGradient<Number>& temperature_gradient)
78  : HeatFlux<Number>(-1.0 * thermal_conductivity.Value() * temperature_gradient.Value()) {}
