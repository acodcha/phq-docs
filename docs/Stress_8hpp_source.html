48 /// \brief Cauchy stress symmetric dyadic tensor. See also PhQ::ScalarStress.

60  /// \brief Constructor. Constructs a stress tensor from a given static pressure using the
61  /// definition of stress due to pressure. Since pressure is compressive, the negative of the
62  /// static pressure contributes to the stress.
63  constexpr explicit Stress(const StaticPressure<Number>& static_pressure)
64  : Stress<Number>(
65  {static_cast<Number>(-1.0) * static_pressure.Value(), static_cast<Number>(0.0),
66  static_cast<Number>(0.0), static_cast<Number>(-1.0) * static_pressure.Value(),
67  static_cast<Number>(0.0), static_cast<Number>(-1.0) * static_pressure.Value()}) {}

183  /// \brief Computes the von Mises stress of this stress tensor using the von Mises yield
184  /// criterion.
185  [[nodiscard]] constexpr ScalarStress<Number> VonMises() const {
186  return ScalarStress<Number>{std::sqrt(
187  0.5
188  * (std::pow(this->value.xx() - this->value.yy(), 2)
189  + std::pow(this->value.yy() - this->value.zz(), 2)
190  + std::pow(this->value.zz() - this->value.xx(), 2)
191  + 6.0
192  * (std::pow(this->value.xy(), 2) + std::pow(this->value.xz(), 2)
193  + std::pow(this->value.yz(), 2))))};
194  }
