48 /// \brief Constitutive model for a compressible Newtonian fluid. This is the simplest constitutive
49 /// model for a compressible fluid. It is similar to the model for an incompressible Newtonian
50 /// fluid, but also includes the effect of the volumetric component of the strain rate tensor in

160  /// \brief Returns the stress resulting from a given strain rate.
161  [[nodiscard]] inline PhQ::Stress<float> Stress(
162  const PhQ::StrainRate<float>& strain_rate) const override {
163  // stress = a * strain_rate + b * trace(strain_rate) * identity_matrix
164  // a = 2 * dynamic_viscosity
165  // b = bulk_dynamic_viscosity
166  const float a{static_cast<float>(2) * static_cast<float>(dynamic_viscosity.Value())};
167  const float b{static_cast<float>(bulk_dynamic_viscosity.Value())
168  * static_cast<float>(strain_rate.Value().Trace())};
169  return PhQ::Stress<float>{
170  SymmetricDyad<float>{a * static_cast<SymmetricDyad<float>>(strain_rate.Value())}
171  + SymmetricDyad<float>{ b, static_cast<float>(0), static_cast<float>(0), b,
172  static_cast<float>(0), b},
173  Standard<Unit::Pressure>
174  };
175  }

237  /// \brief Returns the strain rate resulting from a given stress.
238  [[nodiscard]] inline PhQ::StrainRate<float> StrainRate(
239  const PhQ::Stress<float>& stress) const override {
240  // strain_rate = a * stress + b * trace(stress) * identity_matrix
241  // a = 1 / (2 * dynamic_viscosity)
242  // b = -1 * bulk_dynamic_viscosity /
243  // (2 * dynamic_viscosity * (2 * dynamic_viscosity + 3 * bulk_dynamic_viscosity))
244  const float a{static_cast<float>(1)
245  / (static_cast<float>(2) * static_cast<float>(dynamic_viscosity.Value()))};
246  const float b{
247  static_cast<float>(-bulk_dynamic_viscosity.Value())
248  / (static_cast<float>(2) * static_cast<float>(dynamic_viscosity.Value())
249  * (static_cast<float>(2) * static_cast<float>(dynamic_viscosity.Value())
250  + static_cast<float>(3) * static_cast<float>(bulk_dynamic_viscosity.Value())))};
251  const float c{b * static_cast<float>(stress.Value().Trace())};
252  return PhQ::StrainRate<float>{
253  a * static_cast<SymmetricDyad<float>>(stress.Value())
254  + SymmetricDyad<float>{c, static_cast<float>(0), static_cast<float>(0), c,
255  static_cast<float>(0), c},
256  Standard<Unit::Frequency>
257  };
258  }
