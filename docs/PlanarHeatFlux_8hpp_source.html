46 /// \brief Two-dimensional Euclidean heat flux vector in the XY plane. Contains two components in
47 /// Cartesian coordinates: x and y. For a three-dimensional Euclidean heat flux vector, see
48 /// PhQ::HeatFlux. For scalar heat flux components or for the magnitude of a heat flux vector, see
49 /// PhQ::ScalarHeatFlux.

76  /// \brief Constructor. Constructs a planar heat flux vector from a given scalar thermal
77  /// conductivity and planar temperature gradient vector using Fourier's law of heat conduction.
78  /// Since heat flows opposite the temperature gradient, the resulting heat flux direction is
79  /// opposite the temperature gradient direction.
80  constexpr PlanarHeatFlux(const ScalarThermalConductivity<Number>& scalar_thermal_conductivity,
81  const PlanarTemperatureGradient<Number>& planar_temperature_gradient)

83  -scalar_thermal_conductivity.Value() * planar_temperature_gradient.Value()) {}

85  /// \brief Constructor. Constructs a planar heat flux vector from a given thermal conductivity
86  /// tensor and planar temperature gradient vector using Fourier's law of heat conduction. Since
87  /// heat flows opposite the temperature gradient, the resulting heat flux direction is opposite
88  /// the temperature gradient direction.
89  constexpr PlanarHeatFlux(const ThermalConductivity<Number>& thermal_conductivity,
90  const PlanarTemperatureGradient<Number>& planar_temperature_gradient)
91  : PlanarHeatFlux<Number>(PlanarVector<Number>{Vector<Number>{
92  -1.0 * thermal_conductivity.Value() * planar_temperature_gradient.Value()}}) {}
