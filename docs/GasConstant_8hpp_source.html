46 /// \brief Gas constant of a gas. Not to be confused with the molar gas constant or the specific gas
47 /// constant.
48 template <typename Number = double>
49 class GasConstant : public DimensionalScalar<Unit::HeatCapacity, Number> {
50 public:
51  /// \brief Default constructor. Constructs a gas constant with an uninitialized value.
52  GasConstant() = default;
53 
54  /// \brief Constructor. Constructs a gas constant with a given value expressed in a given heat
55  /// capacity unit.
56  GasConstant(const Number value, const Unit::HeatCapacity unit)
57  : DimensionalScalar<Unit::HeatCapacity, Number>(value, unit) {}
58 
59  /// \brief Constructor. Constructs a gas constant from a given isobaric heat capacity and
60  /// isochoric heat capacity using Mayer's relation.
61  constexpr GasConstant(const IsobaricHeatCapacity<Number>& isobaric_heat_capacity,
62  const IsochoricHeatCapacity<Number>& isochoric_heat_capacity)
63  : GasConstant<Number>(isobaric_heat_capacity.Value() - isochoric_heat_capacity.Value()) {}
64 
65  /// \brief Constructor. Constructs a gas constant from a given isobaric heat capacity and heat
66  /// capacity ratio using the definition of the heat capacity ratio and Mayer's relation.
67  constexpr GasConstant(const HeatCapacityRatio<Number>& heat_capacity_ratio,
68  const IsobaricHeatCapacity<Number>& isobaric_heat_capacity)
69  : GasConstant<Number>(
70  (1.0 - 1.0 / heat_capacity_ratio.Value()) * isobaric_heat_capacity.Value()) {}
71 
72  /// \brief Constructor. Constructs a gas constant from a given isochoric heat capacity and heat
73  /// capacity ratio using the definition of the heat capacity ratio and Mayer's relation.
74  constexpr GasConstant(const HeatCapacityRatio<Number>& heat_capacity_ratio,
75  const IsochoricHeatCapacity<Number>& isochoric_heat_capacity)
76  : GasConstant<Number>((heat_capacity_ratio.Value() - 1.0) * isochoric_heat_capacity.Value()) {}
77 
78  /// \brief Constructor. Constructs a gas constant from a given specific gas constant and mass
79  /// using the definition of the specific gas constant.
80  constexpr GasConstant(
