46 /// \brief Gas constant of a gas. For the mass-specific gas constant, see PhQ::SpecificGasConstant.
47 template <typename NumericType = double>
48 class GasConstant : public DimensionalScalar<Unit::HeatCapacity, NumericType> {
49 public:
50  /// \brief Default constructor. Constructs a gas constant with an uninitialized value.
51  GasConstant() = default;
52 
53  /// \brief Constructor. Constructs a gas constant with a given value expressed in a given heat
54  /// capacity unit.
55  GasConstant(const NumericType value, const Unit::HeatCapacity unit)
56  : DimensionalScalar<Unit::HeatCapacity, NumericType>(value, unit) {}
57 
58  /// \brief Constructor. Constructs a gas constant from a given isobaric heat capacity and
59  /// isochoric heat capacity using Mayer's relation.
60  constexpr GasConstant(const IsobaricHeatCapacity<NumericType>& isobaric_heat_capacity,
61  const IsochoricHeatCapacity<NumericType>& isochoric_heat_capacity)
62  : GasConstant<NumericType>(isobaric_heat_capacity.Value() - isochoric_heat_capacity.Value()) {}
63 
64  /// \brief Constructor. Constructs a gas constant from a given isobaric heat capacity and heat
65  /// capacity ratio using the definition of the heat capacity ratio and Mayer's relation.
66  constexpr GasConstant(const HeatCapacityRatio<NumericType>& heat_capacity_ratio,
67  const IsobaricHeatCapacity<NumericType>& isobaric_heat_capacity)
68  : GasConstant<NumericType>(
69  (1.0 - 1.0 / heat_capacity_ratio.Value()) * isobaric_heat_capacity.Value()) {}
70 
71  /// \brief Constructor. Constructs a gas constant from a given isochoric heat capacity and heat
72  /// capacity ratio using the definition of the heat capacity ratio and Mayer's relation.
73  constexpr GasConstant(const HeatCapacityRatio<NumericType>& heat_capacity_ratio,
74  const IsochoricHeatCapacity<NumericType>& isochoric_heat_capacity)
75  : GasConstant<NumericType>(
76  (heat_capacity_ratio.Value() - 1.0) * isochoric_heat_capacity.Value()) {}
77 
78  /// \brief Constructor. Constructs a gas constant from a given specific gas constant and mass
79  /// using the definition of the specific gas constant.
80  constexpr GasConstant(
81  const SpecificGasConstant<NumericType>& specific_gas_constant, const Mass<NumericType>& mass);

84  ~GasConstant() noexcept = default;

116  /// \brief Statically creates a gas constant with a given value expressed in a given heat capacity
117  /// unit.
118  template <Unit::HeatCapacity Unit>
119  static constexpr GasConstant<NumericType> Create(const NumericType value) {
120  return GasConstant<NumericType>{
121  ConvertStatically<Unit::HeatCapacity, Unit, Standard<Unit::HeatCapacity>>(value)};
122  }
