42 /// @brief Scalar angular acceleration. Represents the time rate of change of an angular speed; see
43 /// also PhQ::AngularSpeed, PhQ::Time, and PhQ::Frequency. Can also represent a component or the
44 /// magnitude of an angular acceleration pseudovector.
45 template <typename Number = double>

55  : DimensionalScalar<Unit::AngularAcceleration, Number>(value, unit) {}
