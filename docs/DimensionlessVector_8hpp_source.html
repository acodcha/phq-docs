43 /// \brief Abstract base class that represents any dimensionless vector physical quantity. Such a
44 /// physical quantity is composed only of a value where the value is a three-dimensional vector.
45 /// Such a physical quantity has no unit of measure and no dimension set.
46 template <typename Number = double>
47 class DimensionlessVector {
48  static_assert(
49  std::is_floating_point<Number>::value,
50  "The Number template parameter of a physical quantity must be a floating-point number type.");
51 
52 public:
53  /// \brief Physical dimension set of this physical quantity. Since this physical quantity is
54  /// dimensionless, its physical dimension set is simply the null set.
55  static constexpr PhQ::Dimensions Dimensions() {
56  return PhQ::Dimensionless;
57  }
58 
59  /// \brief Value of this physical quantity.
60  [[nodiscard]] constexpr const PhQ::Vector<Number>& Value() const noexcept {
61  return value;
62  }

66  return value.Print();
67  }

75  [[nodiscard]] std::string XML() const {
