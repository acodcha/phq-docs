41 /// \brief Abstract base class that represents any dimensionless vector physical quantity. Such a
42 /// physical quantity is composed only of a value where the value is a three-dimensional vector.
43 /// Such a physical quantity has no unit of measure and no dimension set.
44 template <typename NumericType = double>
45 class DimensionlessVector {
46  static_assert(std::is_floating_point<NumericType>::value,
47  "The NumericType template parameter of a physical quantity must be a numeric "
48  "floating-point type: float, double, or long double.");
49 
50 public:
51  /// \brief Physical dimension set of this physical quantity. Since this physical quantity is
52  /// dimensionless, its physical dimension set is simply the null set.
53  [[nodiscard]] static constexpr PhQ::Dimensions Dimensions() {
54  return PhQ::Dimensionless;
55  }
56 
57  /// \brief Value of this physical quantity.
58  [[nodiscard]] constexpr const PhQ::Vector<NumericType>& Value() const noexcept {
59  return value;
60  }

62  /// \brief Prints this physical quantity as a string.

64  return value.Print();
65  }

67  /// \brief Serializes this physical quantity as a JSON message.

77  /// \brief Serializes this physical quantity as a YAML message.
