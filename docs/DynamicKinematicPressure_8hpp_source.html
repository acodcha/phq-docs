44 /// \brief Dynamic kinematic pressure, which is dynamic pressure divided by mass density; see
45 /// PhQ::DynamicPressure and PhQ::MassDensity.
46 template <typename Number = double>
47 class DynamicKinematicPressure : public DimensionalScalar<Unit::SpecificEnergy, Number> {
48 public:
49  /// \brief Default constructor. Constructs a dynamic kinematic pressure with an uninitialized
50  /// value.
51  DynamicKinematicPressure() = default;
52 
53  /// \brief Constructor. Constructs a dynamic kinematic pressure with a given value expressed in a
54  /// given specific energy unit.
55  DynamicKinematicPressure(const Number value, const Unit::SpecificEnergy unit)
56  : DimensionalScalar<Unit::SpecificEnergy, Number>(value, unit) {}
57 
58  /// \brief Constructor. Constructs a dynamic kinematic pressure from a given speed using the
59  /// definition of dynamic kinematic pressure.
60  explicit constexpr DynamicKinematicPressure(const Speed<Number>& speed)
61  : DynamicKinematicPressure<Number>(0.5 * std::pow(speed.Value(), 2)) {}
62 
63  /// \brief Constructor. Constructs a dynamic kinematic pressure from a given total kinematic
64  /// pressure and static kinematic pressure using the definition of total kinematic pressure.
65  constexpr DynamicKinematicPressure(
66  const TotalKinematicPressure<Number>& total_kinematic_pressure,
67  const StaticKinematicPressure<Number>& static_kinematic_pressure);
68 
69  /// \brief Constructor. Constructs a dynamic kinematic pressure from a given dynamic pressure and
70  /// mass density using the definition of dynamic kinematic pressure.
71  constexpr DynamicKinematicPressure(
72  const DynamicPressure<Number>& dynamic_pressure, const MassDensity<Number>& mass_density)
73  : DynamicKinematicPressure<Number>(dynamic_pressure.Value() / mass_density.Value()) {}
74 
75  /// \brief Destructor. Destroys this dynamic kinematic pressure.
76  ~DynamicKinematicPressure() noexcept = default;

107 
108  /// \brief Statically creates a dynamic kinematic pressure of zero.
109  static constexpr DynamicKinematicPressure<Number> Zero() {
110  return DynamicKinematicPressure<Number>{static_cast<Number>(0)};
111  }
112 
113  /// \brief Statically creates a dynamic kinematic pressure with a given value expressed in a given
114  /// specific energy unit.
115  template <Unit::SpecificEnergy Unit>
116  static constexpr DynamicKinematicPressure<Number> Create(const Number value) {
117  return DynamicKinematicPressure<Number>{

119  }

124  }
