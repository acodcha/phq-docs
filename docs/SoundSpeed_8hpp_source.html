53 /// \brief Speed of sound. Applies to any deformable material, including fluids and deformable
54 /// solids. Defined as the ratio of a material's isentropic bulk modulus to its mass density; see
55 /// PhQ::IsentropicBulkModulus and PhQ::MassDensity. The speed of sound also appears in the
56 /// definition of the Mach number; see PhQ::MachNumber and PhQ::Speed.
57 template <typename Number = double>
58 class SoundSpeed : public DimensionalScalar<Unit::Speed, Number> {
59 public:
60  /// \brief Default constructor. Constructs a sound speed with an uninitialized value.
61  SoundSpeed() = default;
62 
63  /// \brief Constructs a sound speed from a given value and speed unit.
64  SoundSpeed(const Number value, const Unit::Speed unit)
65  : DimensionalScalar<Unit::Speed, Number>(value, unit) {}
66 
67  /// \brief Constructs a sound speed from an isentropic bulk modulus and a mass density. This is
68  /// the definition of the sound speed; this relation always holds true.
69  SoundSpeed(const IsentropicBulkModulus<Number>& isentropic_bulk_modulus,
70  const MassDensity<Number>& mass_density)
71  : SoundSpeed<Number>(std::sqrt(isentropic_bulk_modulus.Value() / mass_density.Value())) {}
72 
73  /// \brief Constructs a sound speed from a heat capacity ratio, a static pressure, and a mass
74  /// density. This relation applies only to an ideal gas.
75  SoundSpeed(const HeatCapacityRatio<Number>& heat_capacity_ratio,
76  const StaticPressure<Number>& static_pressure, const MassDensity<Number>& mass_density)
77  : SoundSpeed<Number>(
78  std::sqrt(heat_capacity_ratio.Value() * static_pressure.Value() / mass_density.Value())) {}
79 
80  /// \brief Constructs a sound speed from a heat capacity ratio, a specific gas constant, and a
81  /// temperature. This relation applies only to an ideal gas.
82  SoundSpeed(const HeatCapacityRatio<Number>& heat_capacity_ratio,
83  const SpecificGasConstant<Number>& specific_gas_constant,
84  const Temperature<Number>& temperature)
85  : SoundSpeed<Number>(std::sqrt(
86  heat_capacity_ratio.Value() * specific_gas_constant.Value() * temperature.Value())) {}
