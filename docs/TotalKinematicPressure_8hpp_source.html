43 /// \brief Total kinematic pressure, which is total pressure divided by mass density; see
44 /// PhQ::TotalPressure and PhQ::MassDensity.
45 template <typename Number = double>
46 class TotalKinematicPressure : public DimensionalScalar<Unit::SpecificEnergy, Number> {
47 public:
48  /// \brief Default constructor. Constructs a total kinematic pressure with an uninitialized value.
49  TotalKinematicPressure() = default;
50 
51  /// \brief Constructor. Constructs a total kinematic pressure with a given value expressed in a
52  /// given specific energy unit.
53  TotalKinematicPressure(const Number value, const Unit::SpecificEnergy unit)
54  : DimensionalScalar<Unit::SpecificEnergy, Number>(value, unit) {}
55 
56  /// \brief Constructor. Constructs a total kinematic pressure from a given static kinematic
57  /// pressure and dynamic kinematic pressure using the definition of total kinematic pressure.
58  constexpr TotalKinematicPressure(
59  const StaticKinematicPressure<Number>& static_kinematic_pressure,
60  const DynamicKinematicPressure<Number>& dynamic_kinematic_pressure)
61  : TotalKinematicPressure<Number>(
62  static_kinematic_pressure.Value() + dynamic_kinematic_pressure.Value()) {}
63 
64  /// \brief Constructor. Constructs a total kinematic pressure from a given total pressure and mass
65  /// density using the definition of total kinematic pressure.
66  constexpr TotalKinematicPressure(
67  const TotalPressure<Number>& total_pressure, const MassDensity<Number>& mass_density)
68  : TotalKinematicPressure<Number>(total_pressure.Value() / mass_density.Value()) {}
69 
70  /// \brief Destructor. Destroys this total kinematic pressure.
71  ~TotalKinematicPressure() noexcept = default;

73  /// \brief Copy constructor. Constructs a total kinematic pressure by copying another one.

105  /// \brief Statically creates a total kinematic pressure with a given value expressed in a given
106  /// specific energy unit.

109  return TotalKinematicPressure<Number>{

111  }
