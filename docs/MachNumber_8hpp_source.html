40 /// \brief Mach number of a fluid flow. Measures the local compressibility of a fluid flow.
41 /// Represents the ratio of a fluid's local speed to its local speed of sound. See also PhQ::Speed
42 /// and PhQ::SoundSpeed.
43 template <typename NumericType = double>
44 class MachNumber : public DimensionlessScalar<NumericType> {
45 public:
46  /// \brief Default constructor. Constructs a Mach number with an uninitialized value.
47  MachNumber() = default;

49  /// \brief Constructor. Constructs a Mach number with a given value.
50  explicit constexpr MachNumber(const NumericType value)
51  : DimensionlessScalar<NumericType>(value) {}

53  /// \brief Constructor. Constructs a Mach number from a given speed and sound speed using the
54  /// definition of the Mach number.

56  : MachNumber<NumericType>(speed.Value() / sound_speed.Value()) {}
